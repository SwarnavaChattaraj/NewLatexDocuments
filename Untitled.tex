\section{Claim Box}
\begin{claim}
    If \( n \) is an even integer, then \( n^2 \) is also even.
\end{claim}
\begin{proof}
    Let \( n = 2k \). Then \( n^2 = 4k^2 \), which is even.
\end{proof}

\begin{problem}[A1]
    Find all real solutions to \( x^2 - 5x + 6 = 0 \).
\end{problem}

\section{Solution Box}
\begin{solution}
    Factor the equation:
    \[ (x-2)(x-3) = 0 \]
